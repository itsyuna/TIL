var isValid = function(s) {
    let stack = [];
    let open = {'(' : ')', '{' : '}', '[' : ']'};
    let close = {')':true, '}':true, ']': true};
    
    for (let elem of s) {
        if (open[elem]) {
            stack.push(elem);
        } else if (close[elem]) {
            if (open[stack.pop()] !== elem) return false;
        }
    }
    
    return stack.length === 0;
};
