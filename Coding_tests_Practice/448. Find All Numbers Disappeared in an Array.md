var findDisappearedNumbers = function (nums) {
  let answer = [];

  for (let i = 0; i < nums.length; i++) {
    let idx = Math.abs(nums[i]) - 1;
    if (nums[idx] > 0) {
      nums[idx] *= -1;
    }
  }

  for (let i = 1; i < nums.length; i++) {
    if (nums[i - 1] > 0) {
      answer.push(i);
    }
  }
  return answer;
};
