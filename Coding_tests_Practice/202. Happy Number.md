var isHappy = function(n) {
    let result = false;
    let arr = n.toString().split('').map((n)=> n * n)
    let sum = arr.reduce((a,b) => a + b)
    
    if(sum === 1){
       return true;
    } else if (sum === 4){
       return false;
    } else {
       result = isHappy(sum)
    }
    return result;
};
